Errors

func makeGreeting(u User) (string, error) {
    if !u.canGreet {
        return "", errors.New("cannot greet")
    }

    return fmt.Sprintf("Hello, %s %s", u.FirstName, u.LastName), nil
}

func main() {

    greeting, err := makeGreeting(user)
    if err != nil {
        fmt.Print("cannot greet: ", err)
        return
    }

    fmt.Println(greeting)
}

Channels

package main

import (
	"fmt"
	"math/rand"
	"time"
)


func numbers(ch chan int) {
    for i := 0; i < 5; i++ {
        ch <- i
        time.Sleep(time.Duration(rand.Intn(1000)) * time.Millisecond)
    }
}

func main() {
    ch := make(chan int)
    go numbers(ch)
    for i := 0; i < 5; i++ {
        v := <-ch
        fmt.Println(v)
    }
}

Generator

package main

import (
	"fmt"
	"math/rand"
	"time"
)


func numbers() chan int {
    ch := make(chan int)

    go func() {
        for i := 0; i < 5; i++ {
            ch <- i
            time.Sleep(time.Duration(rand.Intn(1000)) * time.Millisecond)
        }
        close(ch)
    }()

    return ch
}

func main() {
    for v := range numbers() {
        fmt.Println(v)
    }
}
